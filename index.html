// El ID único extraído directamente de la URL de tu Google Sheet const SPREADSHEET_ID = "1_QQpnP6R1qRATDB3a7y4NQVIfVASz45urmOLYlh17lY"; function doGet(e) { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheetJugadores = ss.getSheetByName("Jugadores"); const data = sheetJugadores.getDataRange().getValues(); let jugadores = []; // Omitir encabezados (fila 1) for (let i = 1; i < data.length; i++) { if(data[i][0] && data[i][0].toString().trim() !== "") { jugadores.push({ nombre: data[i][0], jugados: data[i][1] || 0, ganados: data[i][2] || 0, perdidos: data[i][3] || 0 }); } } return ContentService.createTextOutput(JSON.stringify(jugadores)) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ error: error.message })) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheetJugadores = ss.getSheetByName("Jugadores"); const sheetHistorial = ss.getSheetByName("Historial"); const payload = JSON.parse(e.postData.contents); const accion = payload.accion; if (accion === "nuevo_jugador") { sheetJugadores.appendRow([payload.nombre, 0, 0, 0]); return ContentService.createTextOutput(JSON.stringify({status: "ok"})) .setMimeType(ContentService.MimeType.JSON); } if (accion === "guardar_juego") { const fecha = new Date().toLocaleString(); const ganador = payload.ganador; const puntajeGanador = payload.puntajeGanador; const perdedores = payload.perdedores; // Guardar en historial sheetHistorial.appendRow([fecha, ganador, puntajeGanador, perdedores.join(", ")]); // Actualizar estadísticas de jugadores const dataJugadores = sheetJugadores.getDataRange().getValues(); for (let i = 1; i < dataJugadores.length; i++) { let nombreFila = dataJugadores[i][0]; if (nombreFila === ganador) { sheetJugadores.getRange(i + 1, 2).setValue((dataJugadores[i][1] || 0) + 1); // Jugados sheetJugadores.getRange(i + 1, 3).setValue((dataJugadores[i][2] || 0) + 1); // Ganados } else if (perdedores.includes(nombreFila)) { sheetJugadores.getRange(i + 1, 2).setValue((dataJugadores[i][1] || 0) + 1); // Jugados sheetJugadores.getRange(i + 1, 4).setValue((dataJugadores[i][3] || 0) + 1); // Perdidos } } return ContentService.createTextOutput(JSON.stringify({status: "ok"})) .setMimeType(ContentService.MimeType.JSON); } } catch (error) { return ContentService.createTextOutput(JSON.stringify({ error: error.message })) .setMimeType(ContentService.MimeType.JSON); } }
